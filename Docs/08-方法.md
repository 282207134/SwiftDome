## 实例方法 vs 类型方法

```swift
class Counter {
    var count = 0
    func increment() {
        count += 1
    }

    static func reset() {
        print("全局重置")
    }
}
```

## mutating 方法

```swift
struct Point {
    var x = 0.0, y = 0.0

    mutating func moveBy(x deltaX: Double, y deltaY: Double) {
        x += deltaX
        y += deltaY
    }
}
```
